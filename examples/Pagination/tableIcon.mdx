import { Pagination } from "pol-ui";

const PaginationComponent = () => {
  const [page, setPage] = useState(1);

  return (
    <Pagination
      currentPage={page}
      onPageChange={setPage}
      totalPages={100}
      hasLabels={true}
      showIcons={true}

};
